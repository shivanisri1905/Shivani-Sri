create database amazon;
use amazon;
CREATE TABLE category (
     cid int primary key,
     cname varchar(25) NOT NULL
);

insert into category values(1001,'Electronics');
insert into category values(1002,'Furniture');
select * from category;

CREATE TABLE products (
      pin int primary key,
      pname varchar(50) NOT NULL,
      cid int NOT NULL,
      FOREIGN KEY (cid) REFERENCES category (cid)
);
insert into products values(901,'Titan Smart Watch',1001);
insert into products values(902,'Wooden Chair',1002);
insert into products values(903,'IPhone 16s',1001);
select * from products;
